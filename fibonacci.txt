//non recursive
#include <iostream>
using namespace std;

int main() {
    int i, n, t1 = 0, t2 = 1, nT;
    cout<<"Enter number of elements ";
    cin>>n;

    for (i = 1; i <= n; ++i) {
        cout<<t1<<" ";
        nT = t1 + t2;
        t1 = t2;
        t2 = nT;
    }
    return 0;
}


//Recursive fibonacii series
#include<iostream>
using namespace std;
int fibo(int n)
{
    if(n<=1) // base case
        return n;
    return fibo(n-1)+fibo(n-2);
}

int main()
{
int n,i=0;
cout<<"Enter number of elements ";
cin>>n;
while(i<n)
{
cout<<" "<<fibo(i);
i++;
}
return 0;
}
